1. 点击右上角 **"New +"** 按钮
2. 选择 **"Web Service"**
3. 点击 **"Connect a repository"**（如果是第一次）
4. 在仓库列表中找到 **"aye0321123/bazi"**

### 测试 3：Python 测试

**创建测试脚本 `test_my_api.py`：**

```python
import requests
import json

# 替换为你的 Render URL
API_URL = "https://你的地址.onrender.com"

# 读取 Cookie
with open('bazi_credentials.json', 'r', encoding='utf-8') as f:
    COOKIE = json.load(f)['cookie']

# 测试一键对话
print("🧪 测试 Cloud API...")

response = requests.post(
    f"{API_URL}/api/chat",
    headers={
        "X-Bazi-Cookie": COOKIE,
        "Content-Type": "application/json"
    },
    json={"content": "今天运势如何？"}
)

result = response.json()

if result['success']:
    print(f"✅ 成功！")
    print(f"Session ID: {result['session_id']}")
    print(f"官网链接: {result['chat_url']}")
    print(f"\n提示: 访问官网链接查看 AI 回复")
else:
    print(f"❌ 失败: {result['error']}")
```

**运行测试：**

```bash
python test_my_api.py
```

```html
        async function askBazi() {
            const apiUrl = document.getElementById('apiUrl').value.trim();
            const cookie = document.getElementById('cookie').value.trim();

            const resultDiv = document.getElementById('result');
            
            // 验证输入

                return;

            if (!cookie) {
                alert('请输入 Cookie');
                return;

            if (!question) {
                alert('请输入问题');
                return;

            // 显示加载状态

                // 调用 API
                const response = await fetch(`${apiUrl}/api/chat`, {

                        'Content-Type': 'application/json'

                    // 5秒后开始同步
                    setTimeout(() => {
                        syncMessages(apiUrl, cookie, result.session_id);
                    }, 5000);

                    resultDiv.innerHTML = `<div class="loading">❌ 失败: ${result.error}</div>`;
                }
            } catch (error) {
                resultDiv.innerHTML = `<div class="loading">❌ 错误: ${error.message}</div>`;
            }
        }
        
        async function syncMessages(apiUrl, cookie, sessionId) {
            const resultDiv = document.getElementById('result');
            
            try {
                const response = await fetch(
                    `${apiUrl}/api/session/${sessionId}/messages`,
                    {
                        headers: {
                            'X-Bazi-Cookie': cookie
                        }
                    }
                );

                    if (!hasAI) {
                        // 继续同步
                        setTimeout(() => {
                            syncMessages(apiUrl, cookie, sessionId);
                        }, 5000);
                    }

                // 继续尝试同步
                setTimeout(() => {
                    syncMessages(apiUrl, cookie, sessionId);
                }, 5000);
```

## 📊 API 端点

| 端点 | 方法 | 说明 | 认证 |
|------|------|------|------|
| `/` | GET | API 信息 | 否 |
| `/docs` | GET | API 文档 | 否 |
| `/health` | GET | 健康检查 | 否 |
| `/api/session/create` | POST | 创建会话 | 是 |
| `/api/session/<id>/send` | POST | 发送消息 | 是 |
| `/api/session/<id>/messages` | GET | 获取消息 | 是 |
| `/api/chat` | POST | 一键对话（推荐） | 是 |

**认证方式：** 在请求头中添加 `X-Bazi-Cookie: 你的Cookie`

## ⚠️ 重要提示

### 关于 AI 回复

**纯 API 调用无法自动触发 AI 回复！**

**工作流程：**

```
1. 用户输入问题
    ↓
2. 调用 API 发送消息
    ↓
3. 打开官网链接（触发 AI）
    ↓
4. 每 5 秒同步消息
    ↓
5. AI 回复显示在网页
```

**成功率：95%+**（当用户访问官网时）
